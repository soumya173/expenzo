// Creating tables
Table users {
  id int [pk, increment]
  type int [ref: - userTypes.id]
  fname varchar [not null]
  lname varchar
  created_at timestamp [default: `now()`]
  mobile varchar
  email varchar
  password varchar [not null]
  mobile_verified boolean [default: false]
  email_verified boolean [default: false]
}

Table userTypes {
  id int [pk, increment]
  name varchar [not null]
}

Table groups {
  id int [pk, increment]
  name varchar [not null]
  description varchar [default: null]
  type int [ref: - groupTypes.id]
  notify_users boolean [default: false]
}

Table groupTypes {
  id int [pk, increment]
  name varchar [not null]
}

Table groupUserMappings {
  id int [pk, increment]
  group_id int [ref: - groups.id]
  user_id int [ref: - users.id]
}

Table transactions {
  id int [pk, increment]
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
  added_by int [ref: > users.id]
  group_id int [ref: > groups.id]
  amount int [default: 0]
  description varchar [default: null]
  paid_by int [ref: > users.id]
  category_id int [ref: - categories.id]
}

Table categories {
  id int [pk, increment]
  name varchar [not null]
  icon_url varchar [default: null]
}

Table budgets {
  id int [pk, increment]
  name varchar
  group_id int [ref: - groups.id]
  amount int [not null]
  created_at timestamp [default: `now()`]
  updated_at timestamp [default: `now()`]
  created_by int [ref: - users.id]
  applicable_from timestamp [default: `now()`]
  applicable_to timestamp [default: `now()`]
}

Table budgetCategoryWise {
  id int [pk, increment]
  budget_id int [ref: - budgets.id]
  category_id int [ref: - categories.id]
  amount int [not null]
}

Table notifications {
  id int [pk, increment]
  created_at timestamp [default: `now()`]
  description varchar [not null]
  group_id int [ref: > groups.id]
}

Table notificationReadBy {
  id int [pk, increment]
  notification_id int [ref: - notifications.id]
  read_at timestamp [default: `now()`]
  read_by int [ref: - users.id]
}
